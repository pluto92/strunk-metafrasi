\newcommand{\translationsetup}[5]
{
    \def\originalauthor{#1}
    \def\yearpublished{#2}
    \def\translatedtitle{#3}
    \def\maintranslator{#4}

    \pagestyle{texts}

    \chapter[%
        \originalauthor\ -- \emph{\translatedtitle}\ \yearpublished\\%
        {\normalfont\emph{μετάφραση:} \maintranslator}%
    ]{\originalauthor}

    \begin{preface}%
        #5%
    \end{preface}%

    \begin{center}
        \textbf{\translatedtitle}
    
        μετάφραση: \maintranslator
    \end{center}
    
    \noindent
}

\newenvironment{preface}
    {\vspace{-4ex}}
    {\par\noindent\vspace{2ex}\hrulefill\par}
